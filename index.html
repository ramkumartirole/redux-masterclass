Introduction

What is Redux & Why Use It?

Redux is a predictable state container for JavaScript apps. It centralises your application's state so every part of your app can read from and write to a single source of truth.

🗃️ Single Source of Truth

All state lives in one object tree inside a single store. Easier to debug, inspect, and persist.

📖 State is Read-Only

The only way to change state is to emit an action — a plain object describing what happened.

⚙️ Pure Functions

Reducers are pure functions that take previous state + action and return next state. No side effects.

🔮 Predictability

Because state changes follow strict rules, the app behaves consistently and is easy to test.

01-what-is-redux.jsJavaScript
// ─────────────────────────────────────────────
// REDUX IN 3 LINES (conceptually)
// ─────────────────────────────────────────────

// 1. State — the entire app state lives here
const initialState = { count: 0 };

// 2. Action — a plain object describing WHAT happened
const incrementAction = { type: 'counter/increment' };

// 3. Reducer — a PURE function: (state, action) => newState
function reducer(state = initialState, action) {
  switch (action.type) {
    case 'counter/increment':
      return { ...state, count: state.count + 1 };
    default:
      return state;
  }
}

// Store ties everything together
const store = Redux.createStore(reducer);

store.dispatch(incrementAction);
console.log(store.getState()); // { count: 1 }

Store

The single JavaScript object that holds all application state. Created once with createStore(reducer).

Action

A plain JS object with a type field. Describes what happened. Can carry a payload with data.

Reducer

A pure function (state, action) => newState. Never mutates state — always returns a new object.

Dispatch

store.dispatch(action) is the only way to trigger a state change.

Subscribe

store.subscribe(listener) runs your callback every time state changes — used to update the UI.

reducers.jsJavaScript
// ─── Rules of Reducers ──────────────────────────
// 1. Pure function: same input ALWAYS gives same output
// 2. Never mutate state — return new objects/arrays
// 3. Handle all unknown actions by returning current state
// 4. Never produce side effects (API calls, routing, etc.)

const initialState = {
  items: [],
  filter: 'all'
};

function todosReducer(state = initialState, action) {
  switch (action.type) {

    case 'todos/add':
      return {
        ...state,
        items: [...state.items, action.payload]  // ✅ new array
      };

    case 'todos/toggle':
      return {
        ...state,
        items: state.items.map(todo =>
          todo.id === action.payload
            ? { ...todo, completed: !todo.completed }  // ✅ new object
            : todo
        )
      };

    case 'todos/delete':
      return {
        ...state,
        items: state.items.filter(t => t.id !== action.payload)
      };

    case 'todos/setFilter':
      return { ...state, filter: action.payload };

    default:
      return state;  // ← ALWAYS return state for unknown actions
  }
}

// ─── ❌ WRONG — never do this! ──────────────────
function badReducer(state = [], action) {
  if (action.type === 'add') {
    state.push(action.payload);  // ❌ MUTATION! Don't do this
    return state;
  }
}

src/store/counterStore.jsJavaScript
// ─── Action Types ───────────────────────────────
const INCREMENT       = 'counter/increment';
const DECREMENT       = 'counter/decrement';
const INCREMENT_BY    = 'counter/incrementBy';
const RESET           = 'counter/reset';

// ─── Action Creators ────────────────────────────
export const increment    = ()      => ({ type: INCREMENT });
export const decrement    = ()      => ({ type: DECREMENT });
export const incrementBy = (amount) => ({ type: INCREMENT_BY, payload: amount });
export const reset       = ()      => ({ type: RESET });

// ─── Reducer ─────────────────────────────────────
const initialState = { value: 0, history: [] };

function counterReducer(state = initialState, action) {
  switch (action.type) {
    case INCREMENT:
      return { value: state.value + 1, history: [...state.history, state.value + 1] };
    case DECREMENT:
      return { value: state.value - 1, history: [...state.history, state.value - 1] };
    case INCREMENT_BY:
      return { value: state.value + action.payload, history: [...state.history, state.value + action.payload] };
    case RESET:
      return { value: 0, history: [...state.history, 0] };
    default:
      return state;
  }
}

// ─── Store ───────────────────────────────────────
export const counterStore = Redux.createStore(counterReducer);

combineReducers — Splitting State

Real apps have complex state. combineReducers lets each reducer manage its own slice.

src/store/rootReducer.jsJavaScript
import { createStore, combineReducers } from 'redux';

// ─── Slice 1: Counter ────────────────────────────
function counterReducer(state = { value: 0 }, action) {
  switch (action.type) {
    case 'counter/increment': return { value: state.value + 1 };
    case 'counter/decrement': return { value: state.value - 1 };
    default: return state;
  }
}

// ─── Slice 2: Auth ───────────────────────────────
function authReducer(state = { user: null, isLoggedIn: false }, action) {
  switch (action.type) {
    case 'auth/login':
      return { user: action.payload, isLoggedIn: true };
    case 'auth/logout':
      return { user: null, isLoggedIn: false };
    default: return state;
  }
}

// ─── Slice 3: Notifications ──────────────────────
function notifReducer(state = [], action) {
  switch (action.type) {
    case 'notif/add':
      return [...state, { id: Date.now(), ...action.payload }];
    case 'notif/remove':
      return state.filter(n => n.id !== action.payload);
    default: return state;
  }
}

// ─── Root Reducer — combines all slices ──────────
const rootReducer = combineReducers({
  counter:       counterReducer,  // state.counter
  auth:          authReducer,     // state.auth
  notifications: notifReducer,    // state.notifications
});

// Resulting state shape:
// {
//   counter: { value: 0 },
//   auth: { user: null, isLoggedIn: false },
//   notifications: []
// }

export const store = createStore(rootReducer);

Todo App — CRUD with Redux

Add, toggle, delete, filter todos. Demonstrates array state management patterns.

src/store/slices/todosReducer.jsJavaScript
const initialState = {
  items: [],
  filter: 'all',          // 'all' | 'active' | 'completed'
  nextId: 1
};

export function todosReducer(state = initialState, action) {
  switch (action.type) {

    case 'todos/add':
      return {
        ...state,
        items: [...state.items, {
          id: state.nextId,
          text: action.payload,
          completed: false
        }],
        nextId: state.nextId + 1
      };

    case 'todos/toggle':
      return {
        ...state,
        items: state.items.map(t =>
          t.id === action.payload ? { ...t, completed: !t.completed } : t
        )
      };

    case 'todos/delete':
      return { ...state, items: state.items.filter(t => t.id !== action.payload) };

    case 'todos/clearCompleted':
      return { ...state, items: state.items.filter(t => !t.completed) };

    case 'todos/setFilter':
      return { ...state, filter: action.payload };

    default: return state;
  }
}

// ─── Selector (derived state) ────────────────────
export const selectVisibleTodos = (state) => {
  const { items, filter } = state.todos;
  if (filter === 'active')    return items.filter(t => !t.completed);
  if (filter === 'completed') return items.filter(t =>  t.completed);
  return items;
};

src/store/middleware/logger.jsJavaScript
// ─── Middleware Signature ────────────────────────
// A middleware is a curried function:
//   store => next => action => { ... }

// 1. Logger Middleware
const loggerMiddleware = store => next => action => {
  console.group(`Action: ${action.type}`);
  console.log('Prev State:', store.getState());
  console.log('Action:',     action);

  const result = next(action);  // pass to next middleware / reducer

  console.log('Next State:', store.getState());
  console.groupEnd();
  return result;
};

// 2. Analytics Middleware
const analyticsMiddleware = store => next => action => {
  if (action.track) {
    sendToAnalytics({ event: action.type, ...action.meta });
  }
  return next(action);
};

// 3. Auth Guard Middleware
const authMiddleware = store => next => action => {
  const protectedActions = ['cart/checkout', 'user/updateProfile'];
  if (protectedActions.includes(action.type)) {
    const { isLoggedIn } = store.getState().auth;
    if (!isLoggedIn) {
      store.dispatch({ type: 'ui/showLoginModal' });
      return;  // block the action
    }
  }
  return next(action);
};

// ─── Apply Middleware ────────────────────────────
import { createStore, applyMiddleware } from 'redux';

const store = createStore(
  rootReducer,
  applyMiddleware(loggerMiddleware, analyticsMiddleware, authMiddleware)
);

Lesson 6

Async Operations with Thunk

Redux is synchronous by default. redux-thunk lets you dispatch functions (thunks) that can perform async work before dispatching regular actions.

What is a Thunk?

A function that wraps an expression to delay evaluation. In Redux, a thunk is a function that returns another function instead of a plain action object.

Loading States Pattern

Track idle → loading → succeeded/failed to give users feedback during async operations.

src/store/slices/usersReducer.jsJavaScript
// ─── State shape with async status ──────────────
const initialState = {
  data:   [],
  status: 'idle',      // 'idle' | 'loading' | 'succeeded' | 'failed'
  error:  null
};

function usersReducer(state = initialState, action) {
  switch (action.type) {
    case 'users/fetchStart':
      return { ...state, status: 'loading', error: null };
    case 'users/fetchSuccess':
      return { ...state, status: 'succeeded', data: action.payload };
    case 'users/fetchError':
      return { ...state, status: 'failed', error: action.payload };
    default: return state;
  }
}

// ─── Thunk Action Creator ────────────────────────
const fetchUsers = () => async (dispatch, getState) => {
  // dispatch receives: dispatch and getState from the store!
  dispatch({ type: 'users/fetchStart' });

  try {
    const res  = await fetch('https://jsonplaceholder.typicode.com/users');
    const data = await res.json();
    dispatch({ type: 'users/fetchSuccess', payload: data });
  } catch (err) {
    dispatch({ type: 'users/fetchError', payload: err.message });
  }
};

// ─── Conditional Fetch (check state first) ───────
const fetchUsersIfNeeded = () => (dispatch, getState) => {
  const { status } = getState().users;
  if (status === 'idle') dispatch(fetchUsers());
};

// ─── Usage ───────────────────────────────────────
store.dispatch(fetchUsers());          // thunk middleware intercepts this
store.dispatch(fetchUsersIfNeeded());  // only fetches once

Redux Toolkit — The Modern Way

RTK is the official, recommended way to write Redux. It eliminates boilerplate with createSlice, createAsyncThunk, and configureStore.

configureStore

Replaces createStore. Auto-adds redux-thunk, DevTools, and good defaults.

createSlice

Generates action creators AND reducer from one object. Uses Immer so you can write "mutating" code safely.

createAsyncThunk

Handles pending/fulfilled/rejected lifecycle automatically. No manual action types needed.

createSelector

Memoized selectors from Reselect. Prevents unnecessary re-renders in React.

import { createSlice } from '@reduxjs/toolkit';

const counterSlice = createSlice({
  name: 'counter',              // prefix for action types
  initialState: { value: 0, step: 1 },

  reducers: {
    // Immer lets you write "mutating" code — RTK converts it to immutable!
    increment(state) {
      state.value += state.step;  // ✅ mutation is OK inside createSlice
    },
    decrement(state) {
      state.value -= state.step;
    },
    reset(state) {
      state.value = 0;
    },
    setStep(state, action) {       // action.payload is the step amount
      state.step = action.payload;
    },
    incrementByAmount(state, action) {
      state.value += action.payload;
    }
  }
});

  = counterSlice.actions;

// Selector

src/store/slices/usersSlice.jsRedux Toolkit
import { createSlice, createAsyncThunk } from '@reduxjs/toolkit';

// ─── createAsyncThunk ────────────────────────────
// auto-generates: users/fetchAll/pending
//                 users/fetchAll/fulfilled
//                 users/fetchAll/rejected
export const fetchUsers = createAsyncThunk(
  'users/fetchAll',
  async (_, { rejectWithValue }) => {
    try {
      const res = await fetch('https://jsonplaceholder.typicode.com/users');
      if (!res.ok) throw new Error('Network error');
      return await res.json();
    } catch (err) {
      return rejectWithValue(err.message);
    }
  }
);

const usersSlice = createSlice({
  name: 'users',
  initialState: { data: [], status: 'idle', error: null },
  reducers: {},          // sync reducers (empty here)
  extraReducers: (builder) => {
    builder
      .addCase(fetchUsers.pending,   (state) => {
        state.status = 'loading';
      })
      .addCase(fetchUsers.fulfilled, (state, action) => {
        state.status = 'succeeded';
        state.data = action.payload;
      })
      .addCase(fetchUsers.rejected,  (state, action) => {
        state.status = 'failed';
        state.error = action.payload;
      });
  }
});

export default usersSlice.reducer;

Shopping Cart — Real-World Pattern

Combines multiple slices, selectors, derived state, and cross-slice actions into one coherent app.

src/store/slices/cartSlice.jsRedux Toolkit
import { createSlice, createSelector } from '@reduxjs/toolkit';

const cartSlice = createSlice({
  name: 'cart',
  initialState: { items: {}, coupon: null },
  // items: { [productId]: { id, name, price, quantity } }

  reducers: {
    addToCart(state, action) {
      const { id, name, price } = action.payload;
      if (state.items[id]) {
        state.items[id].quantity++;       // Immer: direct mutation OK!
      } else {
        state.items[id] = { id, name, price, quantity: 1 };
      }
    },
    removeFromCart(state, action) {
      delete state.items[action.payload];
    },
    updateQty(state, action) {
      const { id, qty } = action.payload;
      if (qty <= 0) delete state.items[id];
      else state.items[id].quantity = qty;
    },
    applyCoupon(state, action) {
      state.coupon = action.payload;
    },
    clearCart(state) {
      state.items = {};
      state.coupon = null;
    }
  }
});

// ─── Memoized Selectors (createSelector) ─────────
const selectCartItems   = state => Object.values(state.cart.items);
const selectCoupon      = state => state.cart.coupon;

export const selectCartTotal = createSelector(
  [selectCartItems, selectCoupon],
  (items, coupon) => {
    const subtotal = items.reduce((sum, i) => sum + i.price * i.quantity, 0);
    const discount = coupon === 'SAVE10' ? subtotal * 0.1 : 0;
    return { subtotal, discount, total: subtotal - discount };
  }
);

export const selectItemCount = createSelector(
  selectCartItems,
  items => items.reduce((sum, i) => sum + i.quantity, 0)
);

import { useDispatch, useSelector, useStore } from 'react-redux';

// Typed hooks — use these instead of raw useDispatch/useSelector
export const useAppDispatch = () => useDispatch();
export const useAppSelector = useSelector;
export const useAppStore    = useStore;

// ─── Example: domain-specific hook ──────────────
export function useCart() {
  const dispatch = useAppDispatch();
  const items    = useAppSelector(selectCartItems);
  const total    = useAppSelector(selectCartTotal);

  return {
    items, total,
    addItem:    (product) => dispatch(addToCart(product)),
    removeItem: (id)      => dispatch(removeFromCart(id)),
    clear:      ()        => dispatch(clearCart()),
  };
}

// Usage in a component:
function CartIcon() {
  const { items, total, removeItem } = useCart();
  return <span>{items.length} items — ${total.total.toFixed(2)}</span>;
}

✅ DO: Normalise State

Store collections as {ids: [], entities: {}} — not as arrays. Use createEntityAdapter from RTK for free CRUD operations.

✅ DO: Use Selectors

Encapsulate all state access in selector functions. Memoize with createSelector to avoid unnecessary re-renders.

❌ AVOID: Over-Using Redux

Not everything needs Redux. Local UI state (modals, form inputs, hover) should stay in useState.

✅ DO: RTK Query

For data fetching, consider RTK Query — it handles caching, invalidation, loading states, and more automatically.

❌ AVOID: Thunks for Everything

For complex async flows, use RTK Query or Redux-Saga. Keep thunks simple — one concern per thunk.

✅ DO: Install DevTools

The Redux DevTools browser extension gives you time-travel debugging, action replay, and state inspection.
